// Problem 1
  We work for a company building  a smart home thermometer. Our most recent task is this: 
  "Given an  array of tempratures of one day. calculate the temprature amplitude.
   Keep in mind that sometimes there might be a sensor error. 

1)understanding the problem.
  a)what is temprature amplitude
    difference between highest and lowest temprature.
  b)how to calculate max and min temprature.
  c)what is sensor error? what to do with it.

2)Breaking up into sub problems.
  a)how to ignore error.
  b)Find max value in temps array.
  c)Find min value in temps array. 
  d)subtract min from max (amplitude) and return it.
//
//
//
//
//
//
const temps = [3,-2,-6,-1,'error',9,13,17,15,14,9,5];

const calcTempAmplitude = function (temps) {
  let min = temps[0],
    max = temps[0];

  for (let i = 0; i < temps.length; i++) {
     if (typeof temps[i] !== 'number') {
      continue;
    }
    if (temps[i] < min) min = temps[i];

    if (temps[i] > max) max = temps[i];
  }

  console.log(max, min);

  return max - min;
};

console.log(calcTempAmplitude(temps));

Function should recieve two arrays of temps.

solution  
  find functionality two merge two arrays.

const t1 = [4, 1, 0];
const t2 = [3, 9, 1];

const calcTempAmplitudeNew = function (t1, t2) {
  //array concat to add two arrays. (ES5)
  const temps = t1.concat(t2);

  let min = temps[0],
    max = temps[0];

  for (let i = 0; i < temps.length; i++) {
     if (typeof temps[i] !== 'number') {
      continue;
    }
    if (temps[i] < min) min = temps[i];

    if (temps[i] > max) max = temps[i];
  }

  console.log(max, min);

  return max - min;
};

console.log(calcTempAmplitudeNew(t1, t2));
